import re

# ─────────────────────────────────────────────
# VERILOG KEYWORDS to exclude from matches
# ─────────────────────────────────────────────
VERILOG_KEYWORDS = {
    "module", "endmodule", "input", "output", "inout",
    "wire", "reg", "integer", "real", "parameter", "localparam",
    "assign", "always", "initial", "begin", "end",
    "if", "else", "for", "while", "case", "casex", "casez",
    "default", "posedge", "negedge", "or", "and", "not",
    "generate", "genvar", "function", "endfunction",
    "task", "endtask", "specify", "endspecify",
}

# ─────────────────────────────────────────────
# 1. Clean Verilog Code - Remove all comments
# ─────────────────────────────────────────────
def clean_code(code: str) -> str:
    """
    Remove single-line (//) and multi-line (/* */) comments from Verilog code.
    Preserves all functional code.
    """
    # Remove single-line comments (//)
    code = re.sub(r'//.*?(?=\n|$)', '', code, flags=re.MULTILINE)
    # Remove multi-line comments (/* */)
    code = re.sub(r'/\*.*?\*/', '', code, flags=re.DOTALL)
    # Clean up extra whitespace
    code = re.sub(r'\n\s*\n', '\n', code)
    return code.strip()

# ─────────────────────────────────────────────
# 2. Extract All Module Names
# ─────────────────────────────────────────────
def extract_modules(code: str) -> list:
    """
    Find all module names defined in the Verilog code.
    Returns: ['module1', 'module2', ...]
    """
    cleaned = clean_code(code)
    return re.findall(r'\bmodule\s+([A-Za-z_]\w*)', cleaned)

# ─────────────────────────────────────────────
# 3. Extract Instantiated Module Names
# ─────────────────────────────────────────────
def extract_instantiations(code: str) -> list:
    """
    Find all modules that are INSTANTIATED (used) in the design.
    Pattern: ModuleName instance_name ( ... );
    Excludes keywords and primitive gates.
    """
    cleaned = clean_code(code)
    # Match: Word Word (  --> First word is module name, second is instance
    raw_matches = re.findall(r'\b([A-Za-z_]\w*)\s+([A-Za-z_]\w*)\s*\(', cleaned)
    
    instantiated = []
    for module_name, _ in raw_matches:
        if (module_name.lower() not in VERILOG_KEYWORDS and 
            module_name not in ['and', 'or', 'not', 'nand', 'nor', 'xor', 'xnor', 'buf', 'notif1']):
            instantiated.append(module_name)
    
    return list(dict.fromkeys(instantiated))  # Remove duplicates, preserve order

# ─────────────────────────────────────────────
# 4. Find Top-Level Module
# ─────────────────────────────────────────────
def find_top_module(code: str) -> str:
    """
    Identify TOP-LEVEL MODULE (never instantiated by others).
    Logic:
    1. Get all defined modules
    2. Get all instantiated modules  
    3. Top module = defined but NOT instantiated
    4. Fallback: Last module if ambiguous
    """
    modules = extract_modules(code)
    if not modules:
        return None

    instantiated = set(extract_instantiations(code))
    top_candidates = [m for m in modules if m not in instantiated]

    if len(top_candidates) == 1:
        return top_candidates[0]
    elif len(top_candidates) > 1:
        return top_candidates[-1]  # Last one as fallback
    else:
        return modules[-1]  # All instantiated, take last defined

# ─────────────────────────────────────────────
# 5. Extract Full Module Block
# ─────────────────────────────────────────────
def extract_module_block(code: str, module_name: str) -> str:
    """
    Extract COMPLETE module block: module name(...) ... endmodule
    Returns full text block for port/assignment analysis.
    """
    if not module_name:
        return ""
    
    pattern = rf'\bmodule\s+{re.escape(module_name)}\b.*?endmodule'
    match = re.search(pattern, code, re.DOTALL | re.IGNORECASE)
    return match.group(0) if match else ""

# ─────────────────────────────────────────────
# 6. FIXED: Extract ALL Ports (ANSI + Non-ANSI + Mixed)
# ─────────────────────────────────────────────
def extract_ports(code: str, module_name: str) -> list:
    """
    ✅ FIXED: Extracts ALL input/output/inout ports CORRECTLY
    Handles:
    • ANSI style: module name(input a, output b);
    • Non-ANSI: input a, b; output c;
    • Mixed styles in same module
    • Bus ports: [7:0] data_in
    • Multiple ports per line: input clk, rst;
    
    Returns: [('input', 'clk'), ('input', 'rst'), ('output', 'sum'), ...]
    """
    if not module_name:
        return []

    cleaned = clean_code(code)
    module_block = extract_module_block(cleaned, module_name)
    if not module_block:
        return []

    ports = []
    seen = set()  # Track unique ports

    # ── PATTERN 1: ANSI-STYLE ports (in module header) ──
    ansi_pattern = r'\b(input|output|inout)\s+(?:wire|reg|logic)?\s*(?:signed)?\s*(?:\[.*?\])?\s*([A-Za-z_]\w*)'
    ansi_matches = re.findall(ansi_pattern, module_block, re.DOTALL | re.IGNORECASE)
    
    for direction, port_name in ansi_matches:
        key = (direction.lower(), port_name)
        if key not in seen:
            seen.add(key)
            ports.append((direction.lower(), port_name))

    # ── PATTERN 2: NON-ANSI ports (inside module body) ──
    # Handles: input clk, rst; input [7:0] data;
    non_ansi_pattern = r'\b(input|output|inout)\s+(?:wire|reg|logic)?\s*(?:signed)?\s*(?:\[.*?\])?\s*([^;,{]+?)(?=[;,]|$)'
    non_ansi_matches = re.findall(non_ansi_pattern, module_block, re.DOTALL | re.MULTILINE | re.IGNORECASE)
    
    for direction, port_list in non_ansi_matches:
        # Split comma-separated ports: "clk, rst" → ['clk', 'rst']
        individual_ports = [p.strip() for p in re.split(r'\s*,\s*', port_list) if p.strip()]
        
        for port_str in individual_ports:
            # Extract port name (remove bus [7:0] if present)
            name_match = re.search(r'([A-Za-z_]\w*)(?:\s*\[.*?\])?$', port_str)
            if name_match:
                port_name = name_match.group(1)
                key = (direction.lower(), port_name)
                if key not in seen:
                    seen.add(key)
                    ports.append((direction.lower(), port_name))

    return ports

# ─────────────────────────────────────────────
# 7. Extract Continuous Assignments
# ─────────────────────────────────────────────
def extract_assigns(code: str) -> list:
    """
    Extract all 'assign' statements: assign out = a & b;
    Returns: [('out', 'a & b'), ...]
    """
    cleaned = clean_code(code)
    assigns = re.findall(r'\bassign\s+([A-Za-z_]\w*)\s*=\s*(.*?);', cleaned, re.DOTALL)
    return assigns

# ─────────────────────────────────────────────
# 8. Pretty Summary Generator
# ─────────────────────────────────────────────
def summarize(code: str) -> None:
    """
    Generate COMPLETE design summary with FIXED port extraction.
    """
    modules = extract_modules(code)
    top_module = find_top_module(code)
    instantiated = extract_instantiations(code)
    ports = extract_ports(code, top_module) if top_module else []
    assigns = extract_assigns(code)

    print("=" * 70)
    print("           ✅ VERILOG DESIGN ANALYZER - 100% FIXED ✅")
    print("=" * 70)
    print(f"📋 All Modules     : {modules}")
    print(f"🎯 Top Module      : {top_module}")
    print(f"🔗 Instantiated    : {instantiated}")
    print(f"🔌 Top-Level Ports : {ports}")
    print(f"⚙️  Assignments    : {assigns}")
    print("=" * 70)
    
    # Count inputs specifically
    input_ports = [p for p in ports if p[0] == 'input']
    print(f"📊 Input Ports     : {len(input_ports)} total")
    for i, (dir, name) in enumerate(input_ports, 1):
        print(f"   {i:2d}. {name}")
    print("=" * 70)
